Add elements to start, end, or middle without mutation

// Add to beginning
const withFirst = [1, ...numbers];     // [1, 2, 3, 4]

// Add to end

// Add to middle (using slice)

  ...numbers.slice(0, 1),
  'NEW',
  ...numbers.slice(1)
]; // [2, 'NEW', 3, 4]

// Math.max expects individual arguments

ğŸ§¹ Remove Duplicates

Use Set + Spread to get unique values

const withDups = [1, 2, 2, 3, 3, 3, 4];

// Set keeps only unique values
// Spread converts Set back to array
const unique = [...new Set(withDups)];
// [1, 2, 3, 4]
