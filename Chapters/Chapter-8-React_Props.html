What Are Props?

Props (short for "properties") are the mechanism React uses to pass data from a parent component to a child component. Think of props as function arguments for components.

ğŸ¯ Key Concept

Props make components reusable and configurable. Instead of hardcoding values, you pass them as props, allowing the same component to display different data.

ğŸ”„ Props vs State - Key Difference

ğŸ“¥ Props

  • Passed FROM parent TO child
  • Read-only (immutable)
  • Like function arguments
  • Child cannot modify props
  • Changes come from parent

ğŸ”„ State

  • Managed WITHIN a component
  • Mutable via setState
  • Like component memory
  • Component owns and modifies
  • Changes trigger re-render
âš ï¸ Never Mutate Props!
Props are read-only. Never try to modify props directly like props.name = "New Name". This violates React's one-way data flow and will cause bugs.

// Function to create a single student card
function createStudentCard(student) {
  // Step 1: Create container div
  const card = document.createElement('div');
  card.className = 'student-card grade-' + student.grade;
  card.id = 'student-' + student.id;

  // Step 2: Create name element
  const nameEl = document.createElement('h3');
  nameEl.textContent = student.name;
  nameEl.className = 'student-name';

  // Step 3: Create age element
  const ageEl = document.createElement('p');
  ageEl.textContent = 'Age: ' + student.age;
  ageEl.className = 'student-age';

  // Step 4: Create grade element
  const gradeEl = document.createElement('span');
  gradeEl.textContent = 'Grade: ' + student.grade;
  gradeEl.className = 'student-grade';

  // Step 5: Create delete button
  const deleteBtn = document.createElement('button');
  deleteBtn.textContent = 'Delete';
  deleteBtn.className = 'delete-btn';
  
  // Step 6: Add event listener
  deleteBtn.addEventListener('click', function() {
    deleteStudent(student.id);
  });

  // Step 7: Append all elements to card
  card.appendChild(nameEl);
  card.appendChild(ageEl);
  card.appendChild(gradeEl);
  card.appendChild(deleteBtn);

  return card;
}

// Function to render all students
function renderStudents() {
  const container = document.getElementById('student-container');
  
  // Clear existing content
  container.innerHTML = '';
  
  // Create and append each student card
  students.forEach(function(student) {
    const card = createStudentCard(student);
    container.appendChild(card);
  });
}

// Function to delete a student
function deleteStudent(id) {
  // Find index and remove from array
  const index = students.findIndex(s => s.id === id);
  if (index > -1) {
    students.splice(index, 1);
  }
  
  // Re-render the entire list
  renderStudents();
}

// Initial render
renderStudents();

/**
 * Reusable Button Component
 * Props:
 *   - children: Button text/content
 *   - variant: 'primary' | 'danger' | 'success' (styling)
 *   - onClick: Click handler function
 *   - disabled: Boolean to disable button
 */
function Button({ 
  children, 
  variant = 'primary', 
  onClick, 
  disabled = false 
}) {
  // Define styles based on variant prop
  const styles = {
    primary: { background: '#22d3ee', color: '#000' },
    danger: { background: '#ef4444', color: '#fff' },
    success: { background: '#34d399', color: '#000' }
  };

  return (
    <button
      onClick={onClick}
      disabled={disabled}
      style={{
        ...styles[variant],
        padding: '0.5rem 1rem',
        border: 'none',
        borderRadius: '8px',
        cursor: disabled ? 'not-allowed' : 'pointer',
        opacity: disabled ? 0.5 : 1
      }}
    >
      {children}
    </button>
  );
}

export default Button;

import { useState } from 'react';

/**
 * StudentForm Component
 * Props:
 *   - onAddStudent: Callback function to add new student
 */
function StudentForm({ onAddStudent }) {
  // Local state for form inputs
  const [name, setName] = useState('');
  const [age, setAge] = useState('');
  const [grade, setGrade] = useState('A');

  function handleSubmit(e) {
    e.preventDefault();
    
    // Validate inputs
    if (!name.trim() || !age) {
      alert('Please fill in all fields');
      return;
    }

    // Create new student object
    const newStudent = {
      id: Date.now(), // Simple unique ID
      name: name.trim(),
      age: parseInt(age),
      grade,
      subjects: []
    };

    // Call the callback prop to add student
    onAddStudent(newStudent);

    // Reset form
    setName('');
    setAge('');
    setGrade('A');
  }

  return (
    <form onSubmit={handleSubmit} className="student-form">
      <h3>â• Add New Student</h3>
      
      <div className="form-group">
        <label>Name:</label>
        <input
          type="text"
          value={name}
          onChange={(e) => setName(e.target.value)}
          placeholder="Enter student name"
        />
      </div>

      <div className="form-group">
        <label>Age:</label>
        <input
          type="number"
          value={age}
          onChange={(e) => setAge(e.target.value)}
          placeholder="Enter age"
          min="5"
          max="25"
        />
      </div>

      <div className="form-group">
        <label>Grade:</label>
        <select value={grade} onChange={(e) => setGrade(e.target.value)}>
          <option value="A">A - Excellent</option>
          <option value="B">B - Good</option>
          <option value="C">C - Average</option>
          <option value="D">D - Below Average</option>
          <option value="F">F - Failing</option>
