Understanding Event Handling in React
Events
‚ñº

How Events Work in React

React events are similar to DOM events but with some differences. They use camelCase naming and you pass functions as handlers, not strings.

üìù Syntax Comparison: HTML vs React

‚ùå HTML (Don't use in React)

<!-- HTML style - strings -->
<button onclick="handleClick()">
  Click Me
</button>

<input onchange="handleChange()" />
‚Üí

‚úÖ React (Use this!)

{/* React style - functions */}
<button onClick={handleClick}>
  Click Me
</button>

<input onChange={handleChange} />

üéØ Three Ways to Define Event Handlers

  Event Handler Patterns
function EventExamples() {
  // Method 1: Named function (RECOMMENDED)
  const handleClick = () => {
    console.log('Button clicked!');
  };

  // Method 2: Named function with event object
  const handleInputChange = (event) => {
    console.log('Value:', event.target.value);
  };

  // Method 3: Named function with custom parameter
  const handleItemClick = (itemId) => {
    console.log('Item clicked:', itemId);
  };

  return (
    <div>
      {/* Method 1: Pass function reference */}
      <button onClick={handleClick}>
        Click Me
      </button>

      {/* Method 2: Function automatically receives event */}
      <input onChange={handleInputChange} />

      {/* Method 3: Arrow function to pass arguments */}
      <button onClick={() => handleItemClick(123)}>
        Click Item 123
      </button>

      {/* Method 4: Inline arrow function (for simple cases) */}
      <button onClick={() => console.log('Inline!')}>
        Inline Handler
      </button>
    </div>
  );
}
‚ö†Ô∏è Common Mistake: Calling Instead of Passing
onClick={handleClick()} ‚ùå WRONG - This calls the function immediately!
onClick={handleClick} ‚úÖ CORRECT - This passes the function reference.

function InteractiveExample() {

        onChange={handleInputChange}
        placeholder="Type something..."
      />

    </div>
  );
}

import { useState } from 'react';

function Counter({ min = 0, max = 100, initialValue = 0 }) {
  const [count, setCount] = useState(initialValue);
  const [step, setStep] = useState(1);

  // Check bounds before updating
  const handleIncrement = () => {
    const newValue = count + step;
    if (newValue <= max) {
      setCount(newValue);
    }
  };

  const handleDecrement = () => {
    const newValue = count - step;
    if (newValue >= min) {
      setCount(newValue);
    }
  };

  const handleReset = () => setCount(initialValue);
  
  const handleStepChange = (e) => {
    setStep(parseInt(e.target.value) || 1);
  };

  // Determine if buttons should be disabled
  const isAtMax = count >= max;
  const isAtMin = count <= min;

  return (
    <div className="counter">
      <h1>Counter</h1>
      
      {/* Show bounds info */}
      <p className="bounds-info">
        Range: {min} to {max}
      </p>
      
      <div className="count-display">{count}</div>
      
      <div className="buttons">
        <button 
          onClick={handleDecrement}
          disabled={isAtMin}  {/* Disable at min */}
        >
          ‚àí
        </button>
        
        <button onClick={handleReset}>Reset</button>
        
        <button 
          onClick={handleIncrement}
          disabled={isAtMax}  {/* Disable at max */}
        >
          +
        </button>
      </div>
      
      {/* Warning messages */}
      {isAtMax && <p className="warning">Maximum reached!</p>}
      {isAtMin && <p className="warning">Minimum reached!</p>}
      
      <div className="step-control">
        <label>
          Step:
          <input
            type="number"
            value={step}
            onChange={handleStepChange}

üìù Step 4: Complete Styled Counter

  Counter.jsx - Final Version with Styles
import { useState } from 'react';
import './Counter.css';

function Counter({ min = 0, max = 100, initialValue = 0 }) {
  const [count, setCount] = useState(initialValue);
  const [step, setStep] = useState(1);
  const [history, setHistory] = useState([initialValue]);

  // Update count and track history
  const updateCount = (newValue) => {
    if (newValue >= min && newValue <= max) {
      setCount(newValue);
      setHistory(prev => [...prev, newValue].slice(-10)); // Keep last 10
    }
  };

  const increment = () => updateCount(count + step);
  const decrement = () => updateCount(count - step);
  const reset = () => {
    setCount(initialValue);
    setHistory([initialValue]);
  };

  // Calculate percentage for progress bar
  const percentage = ((count - min) / (max - min)) * 100;
  
  // Determine color based on value
  const getColor = () => {
    if (count < 0) return '#ef4444';  // Red for negative
    if (count === 0) return '#94a3b8'; // Gray for zero
    return '#22d3ee';  // Cyan for positive
  };

  return (
    <div className="counter-container">
      <h1>üî¢ Counter</h1>
      
      {/* Progress bar */}
      <div className="progress-bar">
        <div 
          className="progress-fill"
          style={{ width: `${percentage}%` }}
        />
      </div>
      
      {/* Count display */}
      <div 
        className="count-display"
        style={{ color: getColor() }}
      >
        {count}
      </div>
      
      {/* Buttons */}
      <div className="button-group">
        <button
          className="btn btn-decrement"
          onClick={decrement}
          disabled={count <= min}
        >
          ‚àí {step}
        </button>
        
        <button className="btn btn-reset" onClick={reset}>
          ‚Ü∫ Reset
        </button>
        
        <button
          className="btn btn-increment"
          onClick={increment}
          disabled={count >= max}
        >
          + {step}
        </button>
      </div>
      
      {/* Step selector */}
      <div className="step-selector">
        <span>Step:</span>
        {[1, 5, 10, 25].map(s => (
          <button
            key={s}
            className={`step-btn ${step === s ? 'active' : ''}`}
            onClick={() => setStep(s)}
          >
            {s}
          </button>
        ))}
      </div>
      
      {/* History */}
      <div className="history">
        <span>History:</span>
        <div className="history-items">
          {history.map((val, i) => (
            <span key={i} className="history-item">
              {val}
            </span>
          ))}
        </div>
      </div>
    </div>
  );
}

Important: State Update Patterns

‚ö†Ô∏è State Updates You Must Know

State updates in React have some important behaviors you need to understand to avoid bugs and write better code.

üö® Problem: State is Asynchronous

  The Async Problem
function BuggyCounter() {
  const [count, setCount] = useState(0);

  const handleClick = () => {
    // ‚ùå BUG: These don't work as expected!
    setCount(count + 1);  // count is still 0
    setCount(count + 1);  // count is still 0
    setCount(count + 1);  // count is still 0
    
    // Result: count becomes 1, not 3!
    // Why? All three use the same "stale" count value (0)
    
    console.log(count); // Still logs 0! State hasn't updated yet
  };

  return <button onClick={handleClick}>Count: {count}</button>;
}

‚úÖ Solution: Functional Updates

  Correct Way - Functional Updates
function CorrectCounter() {
  const [count, setCount] = useState(0);

  const handleClick = () => {
    // ‚úÖ CORRECT: Use functional update!
    setCount(prev => prev + 1);  // prev = 0 ‚Üí returns 1
    setCount(prev => prev + 1);  // prev = 1 ‚Üí returns 2
    setCount(prev => prev + 1);  // prev = 2 ‚Üí returns 3
    
    // Result: count becomes 3! ‚úÖ
    // Each update receives the LATEST state value
  };

  return <button onClick={handleClick}>Count: {count}</button>;
}

üìù Rule of Thumb

If your new state depends on the previous state, always use functional updates:

setState(prev => newValueBasedOnPrev)

üîÑ Updating Objects in State

  Updating Object State
function UserProfile() {
  const [user, setUser] = useState({
    name: 'John',
    age: 25,
    city: 'Mumbai'
  });

  // ‚ùå WRONG: Mutating state directly
  const wrongUpdate = () => {
    user.age = 26;  // Never do this!
    setUser(user);   // React won't detect the change
  };

  // ‚úÖ CORRECT: Create new object with spread
  const correctUpdate = () => {
    setUser({
      ...user,        // Copy all existing properties
      age: 26         // Override the one you want to change
    });
  };

  // ‚úÖ CORRECT: Functional update for objects
  const incrementAge = () => {
    setUser(prev => ({
      ...prev,
      age: prev.age + 1
    }));
  };

  return (
    <div>
      <p>{user.name}, {user.age}, {user.city}</p>
      <button onClick={incrementAge}>Birthday!</button>
    </div>
  );
}

  Updating Array State
function TodoList() {
  const [items, setItems] = useState(['Learn React', 'Build Projects']);

  // ‚úÖ ADD item
  const addItem = (newItem) => {
    setItems(prev => [...prev, newItem]);
  };

  // ‚úÖ REMOVE item by index
  const removeItem = (index) => {
    setItems(prev => prev.filter((_, i) => i !== index));
  };

  // ‚úÖ UPDATE item at index
  const updateItem = (index, newValue) => {
    setItems(prev => prev.map((item, i) => 
      i === index ? newValue : item
    ));
  };

  // ‚úÖ INSERT at position
  const insertAt = (index, newItem) => {
    setItems(prev => [
      ...prev.slice(0, index),
      newItem,
      ...prev.slice(index)
    ]);
  };

  return (
    <ul>
      {items.map((item, index) => (
        <li key={index}>
          {item}
          <button onClick={() => removeItem(index)}>√ó</button>
        </li>
      ))}
    </ul>
  );
}
‚ö†Ô∏è Never Mutate State Directly!
Don't use: push, pop, splice, sort (on original)
Do use: spread (...), filter, map, slice, concat

import { useState } from 'react';

const MOODS = [
  { emoji: 'üò¢', label: 'Sad', color: '#3b82f6', score: 1 },
  { emoji: 'üòê', label: 'Neutral', color: '#94a3b8', score: 2 },
  { emoji: 'üòä', label: 'Happy', color: '#22c55e', score: 3 },
  { emoji: 'üòÑ', label: 'Excited', color: '#fbbf24', score: 4 },
  { emoji: 'üòç', label: 'Loved', color: '#ec4899', score: 5 },
];

function MoodTracker() {
  const [selectedMood, setSelectedMood] = useState(2);
  const [history, setHistory] = useState([]);

  const currentMood = MOODS[selectedMood];

  const handleMoodSelect = (index) => {
    setSelectedMood(index);
    const entry = {
      mood: MOODS[index],
      timestamp: new Date().toLocaleTimeString(),
      id: Date.now()
    };
    setHistory(prev => [...prev, entry]);
  };

  // Calculate statistics
  const getStats = () => {
    if (history.length === 0) return null;

    // Count each mood
    const moodCounts = history.reduce((acc, entry) => {
      const label = entry.mood.label;
      acc[label] = (acc[label] || 0) + 1;
      return acc;
    }, {});

    // Find most common mood
    const mostCommon = Object.entries(moodCounts)
      .sort((a, b) => b[1] - a[1])[0];

    // Calculate average score
    const avgScore = history.reduce((sum, e) => sum + e.mood.score, 0) / history.length;

    return {
      total: history.length,
      mostCommon: mostCommon[0],
      mostCommonCount: mostCommon[1],
      avgScore: avgScore.toFixed(1),
      moodCounts
    };
  };

  const stats = getStats();

  return (
    <div className="mood-tracker">
      <h1>üòä Mood Tracker</h1>
      
      {/* Current mood */}
      <div className="mood-display">
        <span className="emoji">{currentMood.emoji}</span>
        <span style={{ color: currentMood.color }}>{currentMood.label}</span>
      </div>
      
      {/* Mood buttons */}
      <div className="mood-buttons">
        {MOODS.map((mood, index) => (
          <button
            key={mood.label}
            className={`mood-btn ${index === selectedMood ? 'active' : ''}`}
            onClick={() => handleMoodSelect(index)}
          >
            {mood.emoji}
          </button>
        ))}
      </div>
      
      {/* Statistics */}
      {stats && (
        <div className="stats-section">
          <h3>üìä Today's Stats</h3>
          <div className="stats-grid">
            <div className="stat-card">
              <span className="stat-value">{stats.total}</span>
              <span className="stat-label">Total Entries</span>
            </div>
            <div className="stat-card">
              <span className="stat-value">{stats.mostCommon}</span>
              <span className="stat-label">Most Common</span>
            </div>
            <div className="stat-card">
              <span className="stat-value">{stats.avgScore}/5</span>
              <span className="stat-label">Avg Score</span>
            </div>
          </div>
          
          {/* Mood distribution bar */}
          <div className="mood-distribution">
            {MOODS.map(mood => {
              const count = stats.moodCounts[mood.label] || 0;
              const percent = (count / stats.total) * 100;
              return (
                <div
                  key={mood.label}
                  className="distribution-bar"
                  style={{
                    width: `${percent}%`,
                    backgroundColor: mood.color
                  }}
                  title={`${mood.label}: ${count}`}
                />
              );
            })}
          </div>
        </div>
      )}
      
      {/* History */}
      <div className="history-section">
        <h3>üìú History</h3>
        {history.length === 0 ? (
          <p>No entries yet. Click a mood!</p>
        ) : (
          <div className="history-list">
            {history.slice().reverse().map(entry => (
              <div key={entry.id} className="history-item">
                <span>{entry.mood.emoji}</span>
                <span>{entry.mood.label}</span>
                <span className="time">{entry.timestamp}</span>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}

export default MoodTracker;

‚ú® State & Events Best Practices

‚úÖ Do's

1. Use Functional Updates

// When new state depends on old state
setCount(prev => prev + 1);
setItems(prev => [...prev, newItem]);

2. Name Handlers Clearly

// Good naming convention
const handleClick = () => {};
const handleSubmit = () => {};
const handleInputChange = () => {};

3. Keep State Minimal

// Store only what you need
// Calculate derived values
const total = items.length; // ‚úÖ
// Don't store: const [total, setTotal]

4. Initialize State Properly

const [count, setCount] = useState(0);
const [name, setName] = useState('');
const [items, setItems] = useState([]);
const [user, setUser] = useState(null);

‚ùå Don'ts

1. Don't Mutate State

// ‚ùå WRONG
state.push(newItem);
state.count = 5;

// ‚úÖ RIGHT
setState([...state, newItem]);
setState({...state, count: 5});

2. Don't Call Functions

// ‚ùå WRONG - calls immediately!
onClick={handleClick()}

// ‚úÖ RIGHT - pass reference
onClick={handleClick}
onClick={() => handleClick(arg)}

3. Don't Use Index as Key (Dynamic)

// ‚ùå WRONG for dynamic lists
items.map((item, i) => <div key={i}>)

// ‚úÖ RIGHT - use unique ID
items.map(item => <div key={item.id}>)

4. Don't Read State After Setting

// ‚ùå WRONG - state not updated yet
setCount(5);
console.log(count); // Still old value!
