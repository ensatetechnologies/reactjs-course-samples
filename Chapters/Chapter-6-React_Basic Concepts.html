üé® JSX: The HTML-Flavored Syntax

JSX (JavaScript XML) is a syntax extension that lets you write HTML-like code in JavaScript. It makes React components more readable and intuitive.

App.js 
import React from 'react';

// JSX looks like HTML but is actually JavaScript
function App() {
  const name = "React Developer";
  const currentYear = new Date().getFullYear();
  
  return (
    <div className="app">
      <h1>Hello, {name}!</h1>
      <p>Welcome to React in {currentYear}</p>
      
      {/* Conditional rendering */}
      {currentYear >= 2024 && (
        <span className="badge">Modern React!</span>
      )}
    </div>
  );
}

export default App;

üé£ The useState Hook

The useState hook is the primary way to add state to functional components. It returns an array with two elements: the current state value and a function to update it.

Counter.js 
import React, { useState } from 'react';

function Counter() {
  // Declare state variable with initial value
  const [count, setCount] = useState(0);
  
  // Functions to update state
  const increment = () => setCount(count + 1);
  const decrement = () => setCount(count - 1);
  const reset = () => setCount(0);

  return (
    <div className="counter">
      <h2>Count: {count}</h2>
      <button onClick={decrement}>-</button>
      <button onClick={reset}>Reset</button>
      <button onClick={increment}>+</button>
    </div>
  );
}

‚ö° Handling Events

React events are named using camelCase and pass functions as event handlers, not strings. Events are wrapped in a synthetic event for cross-browser compatibility.

EventExamples.js 
function EventExamples() {
  const [inputValue, setInputValue] = useState('');

  // Click event
  const handleClick = (e) => {
    e.preventDefault();
    alert('Button clicked!');
  };

  // Input change event
  const handleChange = (e) => {
    setInputValue(e.target.value);
  };

  // Keyboard event
  const handleKeyPress = (e) => {
    if (e.key === 'Enter') {
      console.log('Enter pressed!');
    }
  };

  return (
    <div>
      <button onClick={handleClick}>Click Me</button>
      <input
        value={inputValue}
        onChange={handleChange}
        onKeyPress={handleKeyPress}
      />
    </div>
  );
}
üí° Never Mutate State Directly!
Always use the setter function (setState). Don't do: state.value = newValue. Instead do: setState({ ...state, value: newValue })

üîÄ Conditional Rendering

React allows you to render different UI elements based on conditions. There are several patterns for conditional rendering in JSX.

ConditionalRendering.js 
function UserGreeting({ isLoggedIn, user }) {
  // Method 1: if/else (outside JSX)
  if (!isLoggedIn) {
    return <h1>Please sign in</h1>;
  }
  
  return (
    <div>
      <h1>Welcome back, {user.name}!</h1>
      
      {/* Method 2: Ternary operator */}
      <p>
        Status: {user.isPremium ? 'Premium' : 'Free'}
      </p>
      
      {/* Method 3: Logical AND (&&) */}
      {user.notifications > 0 && (
        <span>{user.notifications} new messages</span>
      )}
      
      {/* Method 4: Logical OR for fallbacks */}
      <img src={user.avatar || '/default.png'} />
    </div>
  );
}

üìã Rendering Lists with .map()

The .map() method is the standard way to render lists in React. Each item needs a unique key prop for React's reconciliation algorithm.

PatientList.js 
function PatientList() {
  const patients = [
    { id: 1, name: 'John Doe', room: '101', status: 'stable' },
    { id: 2, name: 'Jane Smith', room: '102', status: 'critical' },
    { id: 3, name: 'Bob Johnson', room: '103', status: 'recovering' }
  ];

  return (
    <ul>
      {patients.map((patient) => (
        <li key={patient.id}>
          <strong>{patient.name}</strong>
          <span>Room {patient.room}</span>
          <span className={\`status-\${patient.status}\`}>
            {patient.status}
          </span>
        </li>
      ))}
    </ul>
  );
}
üí° Why Keys Matter
Keys help React identify which items have changed, been added, or removed. Use unique, stable IDs (like database IDs), not array indices, as keys when the list can be reordered.

üîÑ Understanding Side Effects

Side effects are operations that affect something outside of your component: fetching data, subscriptions, manually changing the DOM, timers, and logging. The useEffect hook lets you perform side effects in functional components.

useEffect Dependency Array Behavior
useEffect(() => { ... }) ‚Üí Runs after EVERY render useEffect(() => { ... }, []) ‚Üí Runs ONLY on mount useEffect(() => { ... }, [dep]) ‚Üí Runs when 'dep' changes useEffect(() => { return cleanup }) ‚Üí Cleanup on unmount

üé£ useEffect Basics

UseEffectBasics.js 
import { useState, useEffect } from 'react';

function EffectDemo() {
  const [count, setCount] = useState(0);

  // Runs ONLY on mount (empty array)
  useEffect(() => {
    console.log('Component mounted!');
  }, []);

  // Runs when 'count' changes
  useEffect(() => {
    document.title = \`Count: \${count}\`;
  }, [count]);

  // Cleanup function
  useEffect(() => {
    const timer = setInterval(() => {
      console.log('Tick');
    }, 1000);

    return () => clearInterval(timer);
  }, []);

  return <button onClick={() => setCount(c => c+1)}>{count}</button>;
}

üåê Data Fetching Pattern

DataFetching.js 
function UserProfile({ userId }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const controller = new AbortController();
    
    async function fetchUser() {
      try {
        setLoading(true);
        const res = await fetch(
          \`/api/users/\${userId}\`,
          { signal: controller.signal }
        );
        const data = await res.json();
        setUser(data);
      } catch (err) {
        if (err.name !== 'AbortError') {
          setError(err.message);
        }
      } finally {
        setLoading(false);
      }
    }
    
    fetchUser();
    return () => controller.abort();
  }, [userId]);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  return <div>{user.name}</div>;
}

üåê Context API

ThemeContext.js 
import { createContext, useContext, useState } from 'react';

// Create Context
const ThemeContext = createContext();

// Provider Component
export function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('light');
  
  const toggleTheme = () => {
    setTheme(t => t === 'light' ? 'dark' : 'light');
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

// Custom Hook
export function useTheme() {
  return useContext(ThemeContext);
}

// Usage in Component
function Header() {
  const { theme, toggleTheme } = useTheme();
  return (
    <header className={theme}>
      <button onClick={toggleTheme}>Toggle Theme</button>
    </header>
  );
}

import { useState, useEffect } from 'react';
import { View, FlatList, Text, ActivityIndicator } from 'react-native';

function MedicineList() {
  const [medicines, setMedicines] = useState([]);
  const [loading, setLoading] = useState(true);

  }

  return (

      data={medicines}
      keyExtractor={(item) => item.id}
      renderItem={({ item }) => (
        <View style={styles.item}>
          <Text>{item.name}</Text>
          <Text>{item.dosage}</Text>
        </View>
      )}
    />
  );
}

}
